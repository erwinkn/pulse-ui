```python
@override
def emit(self, out: list[str]) -> None:
	# Special: ** with unary +/- on left needs parens
	force_left = (
		self.op == "**"
		and isinstance(self.left, Unary)
		and self.left.op in {"-", "+"}
	)
	if force_left:
		out.append("(")
		self.left.emit(out)
		out.append(")")
	else:
		_emit_paren(self.left, self.op, "left", out)
	out.append(" ")
	out.append(self.op)
	out.append(" ")
	_emit_paren(self.right, self.op, "right", out)
```
