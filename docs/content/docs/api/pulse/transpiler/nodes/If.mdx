<PySourceCode >

```python
@override
def emit(self, out: list[str]) -> None:
	out.append("if (")
	self.cond.emit(out)
	out.append(") {\n")
	for stmt in self.then:
		stmt.emit(out)
		out.append("\n")
	out.append("}")
	if self.else_:
		out.append(" else {\n")
		for stmt in self.else_:
			stmt.emit(out)
			out.append("\n")
		out.append("}")
```
