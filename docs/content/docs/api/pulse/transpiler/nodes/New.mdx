```python
@override
def emit(self, out: list[str]) -> None:
	out.append("new ")
	self.ctor.emit(out)
	out.append("(")
	for i, a in enumerate(self.args):
		if i > 0:
			out.append(", ")
		a.emit(out)
	out.append(")")
```

		"t": "new",
		"ctor": self.ctor.render(),
		"args": [a.render() for a in self.args],
	}
