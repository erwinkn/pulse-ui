@override
def emit(self, out: list[str]) -> None:
	if self.op in {"typeof", "await", "void", "delete"}:
		out.append(self.op)
		out.append(" ")
	else:
		out.append(self.op)
	_emit_paren(self.operand, self.op, "unary", out)
